% (un)comment for completion
\renewcommand{\phantom}[1]{#1}

\dfn{}{
	Un \emph{fonction affine} est une fonction $f$ vérifiant pour tout $x\in\R$
		\begin{align*}
			f(x) = \phantom{ax + b,}
		\end{align*}
	où \phantom{\quad$a$\quad} et \phantom{\quad$b$\quad} sont deux paramètres : \phantom{le coefficient directeur et l'ordonnée à l'origine.}
}{def:affine}

\mprop{}{
	Si $f(x) = ax+b$, alors
		\begin{multicols}{2}
		\begin{enumerate}
			\item $f(0) = \phantom{b ;}$
			\item $f(1) - f(0) = \phantom{a.}$
		\end{enumerate}
		\end{multicols}
}{prop:Cf-affine}

\mprop{}{
	La courbe représentative d'une fonction affine est \phantom{une droite non verticale.}
}{prop:Cf-affine}


\dfn{équation de droite}{
	Pour une fonction affine $f(x) = \phantom{\quad ax+b\quad}$, on parle de $\C_f$ comme la « \emph{droite d'équation \phantom{\quad$y=ax+b$\quad}} ».
}{dfn:eq-droite}


\thm{intersection}{
	Un point $P(x_P ; y_P)$ à l'intersection des courbes $\C_f$ et $\C_g$ vérifie
		\begin{align*}
			\phantom{f(x_P) = g(x_P),} && \et && \phantom{y_P = f(x_P) = g(x_P).}
		\end{align*}
}{thm:intersection}


\thm{variations}{
	On distingue trois cas de figure.
		\begin{itemize}
			\item Si \phantom{\quad$a < 0$\quad}, alors $f$ est \phantom{strictement décroissante.}
			\item Si \phantom{\quad$a=0$\quad}, alors $f$ est \phantom{constante.}
			\item Si \phantom{\quad$a>0$\quad}, alors $f$ est \phantom{strictement croissante.}
		\end{itemize}
}{thm:affine-var}
