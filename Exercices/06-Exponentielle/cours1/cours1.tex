\dfn{taux d'évolution moyen}{
	Considérons \hide{$n$} évolutions successives.
	Alors
	\begin{align*}
		CM = \hide{\dfrac{\text{valeur finale}}{\text{valeur initiale}} }
		&& \et &&
		q =\hide{CM^{1/n}}
	\end{align*}
	où \hide{$CM$ est le coefficient multiplicateur global, et $q$ le coefficient multiplicateur moyen.}
}{dfn:taux-moyen}

\ex{}{
	Le PIB de la Corée du Nord est connu depuis 2009 seulement. 
	En 2009, il était estimé à 28 483,5 Won nord-coréens (WNK).
	En 2024, il est estimé à 43 678,9 WNK.
	
	\begin{itemize}
		\item Le coefficient multiplicateur global est \hide{$CM = \frac{43~678,9}{28~483,5} \approx 1,53$.}
		\item Le coefficient multiplicateur \underline{annuel} moyen est donc \hide{$1,53^{1/15} \approx 1,03$.}
		\item Ceci correspond à \hide{une augmentation annuelle moyenne de 3\%.}
	\end{itemize}

}{ex:q-moyen}

\dfn{}{
	Pour $q>0$, la puissance est étendue aux nombres négatifs par les relations
		\begin{align*}
			q^0 = 1 && \et && \hide{q^{-x} = \dfrac1{q^x}.}
		\end{align*}
}{def:exp-neg}

\ex{}{
$
2^{-3} \hide{= \dfrac1{2^3} = \dfrac18}
\qquad\qquad\qquad\qquad
4^{-2} = \hide{\dfrac1{4^2} = \dfrac1{16}}
\qquad\qquad\qquad\qquad
\dfrac1{36} =\hide{ 6^{-2}}
$
}{ex:pow-neg}

\dfn{}{
	Une fonction $g$ est exponentielle de base $q > 0$ dès qu'elle est de la forme
		\[ \hide{g(x) = g(0) \times q^x.} \]
}{def:fonction-exp}

\thm{variations}{
	Pour \hide{$g(x) = g(0) \times q^x$} avec $g(0) > 0$, on distingue deux cas :
		\begin{enumerate}[leftmargin=100pt]
			\item[Si $q>1$ :] alors $g$ est \hide{croissante.}
			\item[Si $q<1$ :] alors $g$ est \hide{décroissante.}
		\end{enumerate}
}{prop:variations}
