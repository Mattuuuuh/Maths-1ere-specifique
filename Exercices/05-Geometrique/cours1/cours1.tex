\dfn{}{
	Une \emph{suite géométrique} est une fonction $u$ vérifiant pour tout $n\in\N$
		\begin{center}
		\hide{ $u(n+1) = q \times u(n)$}
		\end{center}
	qui se lit : \hide{pour passer d'un terme au suivant, on multiplie par la \emph{raison} $q$.}
}{def:affine}

\thm{}{
	Une suite géométrique $u$ de raison $q$ vérifie
		\begin{align*}
			u(n) = \hide{\qquad u(0)\times q^n.\qquad}
		\end{align*}
}{prop:Cf-affine}


\ex{}{ 
	On considère la suite arithmétique $u$ donnée graphiquement par le nuage de points 
		\[ \bigl(0 ; u(0)\bigr), \qquad \bigl(1; u(1)\bigr),\qquad \bigl(2 ; u(2)\bigr),\qquad \bigl(3 ; u(3)\bigr),\qquad \dots \]
	\begin{center}
	\begin{tikzpicture}[>=stealth, scale=1.1]
		\begin{axis}[xmin = 0, xmax=3.5, ymin=0, ymax=15.8, axis x line=middle, axis y line=middle, axis line style=->, xlabel={rang $n$}, x label style={anchor=north}, grid=both, ytick distance=1, xtick distance = 1, x=3.5cm, extra x ticks = {0}, extra y ticks = {0},]
			\addplot[black, thick, only marks, mark=*] coordinates {(0,4) (1, 6) (2,9) (3,13.5)}
				node[pos=0, left=25pt] {$u$};
		\end{axis}
	\end{tikzpicture}
	\end{center}
	\vspace{15pt}
	$u(n) = \hide{4 \times (1,5)^n}$ \hide{car le terme initial de $u$ est $u(0)=4$, et que pour passer d'un terme à l'autre, on multiplie par $1,5$.}
}{}
