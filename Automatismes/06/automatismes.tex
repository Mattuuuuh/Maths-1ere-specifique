\subsection{Évolution}

\begin{frame}{1\vspace{-32pt}}
	Pour augmenter une valeur de 50\%, il faut
		\begin{multicols}{2}
		\begin{enumerate}
			\item lui ajouter $1,5$
			\item la multiplier par $1,5$
			\item lui ajouter $0,5$
			\item la multiplier par $0,5$
		\end{enumerate}
		\end{multicols}
\end{frame}

\begin{frame}{2\vspace{-32pt}}
	Pour diminuer une valeur de 20\%, il faut
		\begin{multicols}{2}
		\begin{enumerate}
			\item la multiplier par $1,2$
			\item la multiplier par $0,2$
			\item la multiplier par $0,8$
			\item lui soustraire sa moitié
		\end{enumerate}
		\end{multicols}
\end{frame}

\subsection{Suites géométriques}

\begin{frame}{3\vspace{-32pt}}
	Soit $u$ une suite géométrique de raison $2$ et de terme initial $u(0) = 4$.
	
	Calculer $u(1)$.
\end{frame}

\begin{frame}{4\vspace{-32pt}}
	Soit $u$ une suite géométrique de raison $2$ telle que $u(1) = 10$.
	
	Calculer $u(0)$.
\end{frame}
