\subsection{Suites géométriques}

\begin{frame}{1}
    Soit $P$ une suite géométrique de raison $7$ et de terme initial
        \[ P(0) = 3. \]
    Donner $P(1)$.
\end{frame}

\begin{frame}{2}
    Soit $f$ une suite géométrique de raison $11$ vérifiant
        \[ f(12) = 33. \]
    Donner $f(11)$.
\end{frame}

\begin{frame}{3}

\end{frame}
