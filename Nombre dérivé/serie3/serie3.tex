\begin{theorem}[label=thm:der-mono]{Dérivée des monômes usuels}{}
	
	Pour chaque monôme suivante, on donne sa dérivée formelle.

\centering
	\begin{tabular}{|c|c|c|c|c|}\hline
		Monôme & 1 & $x$ & $x^2$ & $x^3$ \\ \hline
		Dérivée & 0 & 1 & $2x$ & $3x^2$ \\ \hline
	\end{tabular}
\end{theorem}

\begin{theorem}[label=thm:lin-der]{Linéarité de la dérivée}{}
	Soient $f, g$ deux fonctions dérivables sur $\R$, et $\kappa\in\R$ un nombre réel quelconque. 
	Alors
		\begin{enumerate}
			\item $(f+g)' = f' + g'$ ; et
			\item $(\kappa f)' = \kappa f'$.
		\end{enumerate}
\end{theorem}
