\exe{
	Soit $f(t) = \sin(180t)$ la fonction donnant la position d'un ressort au temps $t$.
	\footnote{On pourra exprimer $t$ en secondes et $f(t)$ en décimètres si souhaité.}
	On pose $V(t)$ la vitesse instantanée du ressort au temps $t$.
	
	Le but de l'exercice est de trouver $V(1)$, la vitesse instantanée du ressort en $t=1$.
	
	\begin{enumerate}
		\item Remplir le tableau de valeurs figure \ref{fig:f(t)} et esquisser la courbe représentative $\C_f$ de $f$ dans le repère figure \ref{fig:Cf}.
		\item Remplir le tableau de variations de $f$ et de signes de $V$ figure \ref{fig:var-signe}. Quel est le signe de $V(1)$ ?
		\item Calculer les vitesses moyennes $v_h$ entre $t=1$ et $1+h$
			\[ v_h = \dfrac{f(1+h)-f(1)}{h} \]
		pour $h$ de plus en plus petit et remplir le tableau de valeurs figure \ref{fig:vh}.
		\item Faire une hypothèse sur la valeur de $V(1)$: vers quelle valeur $v_h$ semble converger lorsque $h$ tend vers $0$ ?
	\end{enumerate}
}{}
