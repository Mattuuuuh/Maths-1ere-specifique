\exe{
	On considère la fonction $f$ du troisième degré définie sur le domaine $\D=\left[-1 ; 2\right]$ :
		\[ f(x) = 2x^3 - \dfrac12 x^2 -2x + \dfrac25. \]
		
	\begin{enumerate}
		\item Donner $f'$, la fonction dérivée de $f$.
		\item Montrer que $f'(x) = (2x+1)(3x-2)$ en développant soigneusement la forme factorisée.
		\item Remplir le tableau de signes et de variations de la figure \ref{fig:var-signe}.
		\item Compléter les variations avec les images de $f$ aux bornes de $\D$ ainsi qu'aux extrema locaux.
		\item En déduire le maximum et le minimum de $f$ sur $\D$ et les antécédents qui les réalisent.
	\end{enumerate}
}{}

\exe{
	On souhaite étudier les variations de la fonction
		\[ f(x) = x\sqrt{4-x^2}, \]
	sur le domaine $\D = [0; 2[$.
	
	Il est admis que la dérivée de $f$ est donnée par
		\[ f'(x) = \dfrac{4-2x^2}{\sqrt{4-x^2}}. \]
	
	\begin{enumerate}
		\item Montrer que $4-2x^2 = -2(x-\sqrt{2})(x+\sqrt{2})$ en développant soigneusement la forme factorisée.
		\item Remplir le tableau de signes et de variations de la figure \ref{fig:var-signe2}.
		\item En déduire le maximum de $f$ sur $\D$ ainsi que l'antécédent qui le réalise.
	\end{enumerate}
}{}
